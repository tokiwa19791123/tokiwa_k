点滅する文字

._blink {
  animation: blink 1s infinite linear forwards;
  animation-delay: 0s;
}

@keyframes blink {
  0% {
    opacity: 0.0;
  }
  50% {
    opacity: 1.0;
  }
  100% {
    opacity: 0.0;
  }
}

右から流れてくる文字

._marquee {
  position: relative;
  overflow: hidden;
  line-height: 2.0em;
  height: 2.0em;
}
._marquee ._marquee_inner {
  display: inline-block;
  position: absolute;
  top: 0;
  bottom: 0;
  animation: marquee 5s infinite linear forwards;
  animation-delay: 0s;
}

@keyframes marquee {
  0% {
    left: 100%;
    right: -100%;
  }
  100% {
    left: -100%;
    right: 100%;
  }
}

左から流れてくる文字

._marquee_reverse {
  position: relative;
  overflow: hidden;
  line-height: 2.0em;
  height: 2.0em;
}
._marquee_reverse ._marquee_inner {
  display: inline-block;
  position: absolute;
  top: 0;
  bottom: 0;
  text-align: right;
  animation: marquee_reverse 5s infinite linear forwards;
  animation-delay: 0s;
}

@keyframes marquee_reverse {
  0% {
    left: -100%;
    right: 100%;
  }
  100% {
    left: 100%;
    right: -100%;
  }
}
